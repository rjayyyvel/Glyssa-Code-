    <script>
        const flowers = [
            "rose.jpg",
            "lily.jpg",
            "daisy.jpg",
            "orchid.jpg",
            "sunflower.jpg",
            "tulips.gif"
        ];
        let currentFlowerIndex = 0;

        function checkFlower(isYes) {
            const responseDiv = document.getElementById("wrongFlowerResponse");
            const finalMessageDiv = document.getElementById("finalMessage");
            const celebrationImages = document.getElementById("celebrationImages");
            const noButton = document.getElementById("noButton");
            
            responseDiv.style.display = "none";
            finalMessageDiv.style.display = "none";
            celebrationImages.style.display = "none";

            if (isYes && currentFlowerIndex === flowers.length - 1) {
                alert("Yes, Tulips is your favorite flower!");
                finalMessageDiv.style.display = "block";

                setTimeout(() => {
                    celebrationImages.style.display = "block";
                }, 10000);
                return;
            }

            if (isYes) {
                responseDiv.style.display = "block";
                return;
            }

            currentFlowerIndex++;

            if (currentFlowerIndex < flowers.length) {
                const flowerDivs = document.querySelectorAll(".flower");
                flowerDivs.forEach((flower, index) => {
                    flower.classList.remove("visible");
                });

                flowerDivs[currentFlowerIndex].classList.add("visible");
            } else {
                alert("No more flowers to check!");
            }

            if (currentFlowerIndex === flowers.length - 1) {
                noButton.style.display = "none";
            }
        }
    </script>
